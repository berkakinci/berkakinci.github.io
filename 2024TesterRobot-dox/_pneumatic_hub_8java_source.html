31 DataStore(int module) {
32 m_handle = REVPHJNI.initialize(module);
33 m_module = module;
34 m_handleMap.put(module, this);
35
36 final REVPHVersion version = REVPHJNI.getVersion(m_handle);
37 final String fwVersion =
38 version.firmwareMajor + "." + version.firmwareMinor + "." + version.firmwareFix;
39
40 if (version.firmwareMajor > 0 && RobotBase.isReal()) {
41 // Write PH firmware version to roboRIO
42 final String fileName = "REV_PH_" + String.format("%02d", module) + "_WPILib_Version.ini";
43 final File file = new File("/tmp/frc_versions/" + fileName);
44 try {
45 if (file.exists() && !file.delete()) {
46 throw new IOException("Failed to delete " + fileName);
47 }
48
49 if (!file.createNewFile()) {
50 throw new IOException("Failed to create new " + fileName);
51 }
52
53 try (OutputStream output = Files.newOutputStream(file.toPath())) {
54 output.write("[Version]\n".getBytes(StandardCharsets.UTF_8));
55 output.write("model=REV PH\n".getBytes(StandardCharsets.UTF_8));
56 output.write(
57 ("deviceID=" + Integer.toHexString(0x9052600 | module) + "\n")
58 .getBytes(StandardCharsets.UTF_8));
59 output.write(("currentVersion=" + fwVersion).getBytes(StandardCharsets.UTF_8));
60 }
61 } catch (IOException ex) {
62 DriverStation.reportError("Could not write " + fileName + ": " + ex, ex.getStackTrace());
63 }
64 }
65
66 // Check PH firmware version
67 if (version.firmwareMajor > 0 && version.firmwareMajor < 22) {
68 throw new IllegalStateException(
69 "The Pneumatic Hub has firmware version "
70 + fwVersion
71 + ", and must be updated to version 2022.0.0 or later "
72 + "using the REV Hardware Client.");
73 }
74 }

97 private static DataStore getForModule(int module) {
98 synchronized (m_handleLock) {
99 Integer moduleBoxed = module;
100 DataStore pcm = m_handleMap.get(moduleBoxed);
101 if (pcm == null) {
102 pcm = new DataStore(module);
103 }
104 pcm.addRef();

116 private static double voltsToPsi(double sensorVoltage, double supplyVoltage) {
117 return 250 * (sensorVoltage / supplyVoltage) - 25;
118 }
119
121 private static double psiToVolts(double pressure, double supplyVoltage) {
122 return supplyVoltage * (0.004 * pressure + 0.1);

281 @Override
282 public void enableCompressorAnalog(double minPressure, double maxPressure) {
283 if (minPressure >= maxPressure) {
284 throw new IllegalArgumentException("maxPressure must be greater than minPressure");
285 }
286 if (minPressure < 0 || minPressure > 120) {
287 throw new IllegalArgumentException(
288 "minPressure must be between 0 and 120 PSI, got " + minPressure);
289 }
290 if (maxPressure < 0 || maxPressure > 120) {
291 throw new IllegalArgumentException(
292 "maxPressure must be between 0 and 120 PSI, got " + maxPressure);
293 }
294
295 // Send the voltage as it would be if the 5V rail was at exactly 5V.
296 // The firmware will compensate for the real 5V rail voltage, which
297 // can fluctuate somewhat over time.
298 double minAnalogVoltage = psiToVolts(minPressure, 5);
299 double maxAnalogVoltage = psiToVolts(maxPressure, 5);
300 REVPHJNI.setClosedLoopControlAnalog(m_handle, minAnalogVoltage, maxAnalogVoltage);
301 }
