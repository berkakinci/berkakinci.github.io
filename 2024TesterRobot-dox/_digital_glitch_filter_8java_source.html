23 @SuppressWarnings("this-escape")
24 public DigitalGlitchFilter() {
25 m_mutex.lock();
26 try {
27 int index = 0;
28 while (m_filterAllocated[index] && index < m_filterAllocated.length) {
29 index++;
30 }
31 if (index != m_filterAllocated.length) {
32 m_channelIndex = index;
33 m_filterAllocated[index] = true;
34 HAL.report(tResourceType.kResourceType_DigitalGlitchFilter, m_channelIndex + 1, 0);
35 SendableRegistry.addLW(this, "DigitalGlitchFilter", index);
36 }
37 } finally {
38 m_mutex.unlock();
39 }
40 }
41
42 @Override
43 public void close() {
44 SendableRegistry.remove(this);
45 if (m_channelIndex >= 0) {
46 m_mutex.lock();
47 try {
48 m_filterAllocated[m_channelIndex] = false;
49 } finally {
50 m_mutex.unlock();
51 }
52
53 m_channelIndex = -1;
54 }
55 }

57 private static void setFilter(DigitalSource input, int channelIndex) {
58 if (input != null) { // Counter might have just one input
59 // analog triggers are not supported for DigitalGlitchFilters
60 if (input.isAnalogTrigger()) {
61 throw new IllegalStateException("Analog Triggers not supported for DigitalGlitchFilters");
62 }
63 DigitalGlitchFilterJNI.setFilterSelect(input.getPortHandleForRouting(), channelIndex);
64
65 int selected = DigitalGlitchFilterJNI.getFilterSelect(input.getPortHandleForRouting());
66 if (selected != channelIndex) {
67 throw new IllegalStateException(
68 "DigitalGlitchFilterJNI.setFilterSelect(" + channelIndex + ") failed -> " + selected);
69 }
70 }
