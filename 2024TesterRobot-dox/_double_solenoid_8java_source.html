59 @SuppressWarnings({"PMD.UseTryWithResources", "this-escape"})
60 public DoubleSolenoid(
61 final int module,
62 final PneumaticsModuleType moduleType,
63 final int forwardChannel,
64 final int reverseChannel) {
65 m_module = PneumaticsBase.getForType(module, moduleType);
66 boolean allocatedSolenoids = false;
67 boolean successfulCompletion = false;
68
69 m_forwardChannel = forwardChannel;
70 m_reverseChannel = reverseChannel;
71
72 m_forwardMask = 1 << forwardChannel;
73 m_reverseMask = 1 << reverseChannel;
74 m_mask = m_forwardMask | m_reverseMask;
75
76 try {
77 if (!m_module.checkSolenoidChannel(forwardChannel)) {
78 throw new IllegalArgumentException("Channel " + forwardChannel + " out of range");
79 }
80
81 if (!m_module.checkSolenoidChannel(reverseChannel)) {
82 throw new IllegalArgumentException("Channel " + reverseChannel + " out of range");
83 }
84
85 int allocMask = m_module.checkAndReserveSolenoids(m_mask);
86 if (allocMask != 0) {
87 if (allocMask == m_mask) {
88 throw new AllocationException(
89 "Channels " + forwardChannel + " and " + reverseChannel + " already allocated");
90 } else if (allocMask == m_forwardMask) {
91 throw new AllocationException("Channel " + forwardChannel + " already allocated");
92 } else {
93 throw new AllocationException("Channel " + reverseChannel + " already allocated");
94 }
95 }
96 allocatedSolenoids = true;
97
98 HAL.report(
99 tResourceType.kResourceType_Solenoid, forwardChannel + 1, m_module.getModuleNumber() + 1);
100 HAL.report(
101 tResourceType.kResourceType_Solenoid, reverseChannel + 1, m_module.getModuleNumber() + 1);
102 SendableRegistry.addLW(this, "DoubleSolenoid", m_module.getModuleNumber(), forwardChannel);
103 successfulCompletion = true;
104 } finally {
105 if (!successfulCompletion) {
106 if (allocatedSolenoids) {
107 m_module.unreserveSolenoids(m_mask);
108 }
109 m_module.close();
110 }
111 }
112 }
