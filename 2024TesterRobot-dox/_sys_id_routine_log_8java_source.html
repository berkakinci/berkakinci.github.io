96 public MotorLog value(String name, double value, String unit) {
97 var motorEntries = m_logEntries.get(m_motorName);
98 var entry = motorEntries.get(name);
99
100 if (entry == null) {
101 var log = DataLogManager.getLog();
102
103 entry = new DoubleLogEntry(log, name + "-" + m_motorName + "-" + m_logName, unit);
104 motorEntries.put(name, entry);
105 }
106
107 entry.append(value);
108 return this;
109 }
