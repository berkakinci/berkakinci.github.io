49 private void init() {
50 m_analogTrigger = new AnalogTrigger(m_analogInput);
51 m_counter = new Counter();
52
53 m_simDevice = SimDevice.create("AnalogEncoder", m_analogInput.getChannel());
54
55 if (m_simDevice != null) {
56 m_simPosition = m_simDevice.createDouble("Position", Direction.kInput, 0.0);
57 m_simAbsolutePosition = m_simDevice.createDouble("absPosition", Direction.kInput, 0.0);
58 }
59
60 // Limits need to be 25% from each end
61 m_analogTrigger.setLimitsVoltage(1.25, 3.75);
62 m_counter.setUpSource(m_analogTrigger, AnalogTriggerType.kRisingPulse);
63 m_counter.setDownSource(m_analogTrigger, AnalogTriggerType.kFallingPulse);
64
65 SendableRegistry.addLW(this, "Analog Encoder", m_analogInput.getChannel());
66 }

68 private boolean doubleEquals(double a, double b) {
69 double epsilon = 0.00001d;
70 return Math.abs(a - b) < epsilon;
71 }

80 public double get() {
81 if (m_simPosition != null) {
82 return m_simPosition.get();
83 }
84
85 // As the values are not atomic, keep trying until we get 2 reads of the same
86 // value. If we don't within 10 attempts, warn
87 for (int i = 0; i < 10; i++) {
88 double counter = m_counter.get();
89 double pos = m_analogInput.getVoltage();
90 double counter2 = m_counter.get();
91 double pos2 = m_analogInput.getVoltage();
92 if (counter == counter2 && doubleEquals(pos, pos2)) {
93 pos = pos / RobotController.getVoltage5V();
94 double position = counter + pos - m_positionOffset;
95 m_lastPosition = position;
96 return position;
97 }
98 }
99
100 DriverStation.reportWarning(
101 "Failed to read Analog Encoder. Potential Speed Overrun. Returning last value", false);
102 return m_lastPosition;
103 }
