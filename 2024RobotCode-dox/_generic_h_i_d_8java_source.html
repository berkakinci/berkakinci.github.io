456 public void setRumble(RumbleType type, double value) {
457 value = MathUtil.clamp(value, 0, 1);
458 short rumbleValue = (short) (value * 65535);
459
460 switch (type) {
461 case kLeftRumble:
462 this.m_leftRumble = rumbleValue;
463 break;
464 case kRightRumble:
465 this.m_rightRumble = rumbleValue;
466 break;
467 case kBothRumble:
468 default:
469 this.m_leftRumble = rumbleValue;
470 this.m_rightRumble = rumbleValue;
471 break;
472 }
473
474 DriverStationJNI.setJoystickOutputs(
475 (byte) this.m_port, this.m_outputs, this.m_leftRumble, this.m_rightRumble);
476 }
