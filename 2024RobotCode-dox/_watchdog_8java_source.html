97 public void setTimeout(double timeoutSeconds) {

99 m_tracer.clearEpochs();

101 m_queueMutex.lock();

103 m_timeoutSeconds = timeoutSeconds;
104 m_isExpired = false;
105
106 m_watchdogs.remove(this);
107 m_expirationTimeSeconds = m_startTimeSeconds + m_timeoutSeconds;
108 m_watchdogs.add(this);

110 } finally {
111 m_queueMutex.unlock();
112 }

211 @SuppressWarnings("resource")
212 private static void updateAlarm() {
213 if (m_watchdogs.isEmpty()) {
214 NotifierJNI.cancelNotifierAlarm(m_notifier);
215 } else {
216 NotifierJNI.updateNotifierAlarm(
217 m_notifier, (long) (m_watchdogs.peek().m_expirationTimeSeconds * 1e6));
218 }
219 }
220
221 private static Thread startDaemonThread(Runnable target) {
222 Thread inst = new Thread(target);
223 inst.setDaemon(true);
224 inst.start();
225 return inst;
226 }
227
228 private static void schedulerFunc() {
229 while (!Thread.currentThread().isInterrupted()) {
230 long curTime = NotifierJNI.waitForNotifierAlarm(m_notifier);
231 if (curTime == 0) {
232 break;
233 }
234
235 m_queueMutex.lock();
236 try {
237 if (m_watchdogs.isEmpty()) {
238 continue;
239 }
240
241 // If the condition variable timed out, that means a Watchdog timeout
242 // has occurred, so call its timeout function.
243 Watchdog watchdog = m_watchdogs.poll();
244
245 double now = curTime * 1e-6;
246 if (now - watchdog.m_lastTimeoutPrintSeconds > kMinPrintPeriodMicroS) {
247 watchdog.m_lastTimeoutPrintSeconds = now;
248 if (!watchdog.m_suppressTimeoutMessage) {
249 DriverStation.reportWarning(
250 String.format("Watchdog not fed within %.6fs\n", watchdog.m_timeoutSeconds), false);
251 }
252 }
253
254 // Set expiration flag before calling the callback so any
255 // manipulation of the flag in the callback (e.g., calling
256 // Disable()) isn't clobbered.
257 watchdog.m_isExpired = true;
258
259 m_queueMutex.unlock();
260 watchdog.m_callback.run();
261 m_queueMutex.lock();
262
263 updateAlarm();
264 } finally {
265 m_queueMutex.unlock();
266 }
267 }
268 }
