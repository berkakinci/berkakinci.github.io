30public class Ultrasonic implements Sendable, AutoCloseable {
31 // Time (sec) for the ping trigger pulse.
32 private static final double kPingTime = 10 * 1e-6;
33 private static final double kSpeedOfSoundInchesPerSec = 1130.0 * 12.0;
34 // ultrasonic sensor list
35 private static final List<Ultrasonic> m_sensors = new ArrayList<>();
36 // automatic round robin mode
37 private static volatile boolean m_automaticEnabled;
38 private DigitalInput m_echoChannel;
39 private DigitalOutput m_pingChannel;
40 private final boolean m_allocatedChannels;
41 private boolean m_enabled;
42 private Counter m_counter;
43 // task doing the round-robin automatic sensing
44 private static Thread m_task;
45 private static int m_instances;
46
47 private SimDevice m_simDevice;
48 private SimBoolean m_simRangeValid;
49 private SimDouble m_simRange;

60 private static class UltrasonicChecker extends Thread {
61 @Override
62 public synchronized void run() {
63 while (m_automaticEnabled) {
64 for (Ultrasonic sensor : m_sensors) {
65 if (!m_automaticEnabled) {
66 break;
67 }
68
69 if (sensor.isEnabled()) {
70 sensor.m_pingChannel.pulse(kPingTime); // do the ping
71 }
72
73 Timer.delay(0.1); // wait for ping to return
74 }
75 }
76 }
77 }

85 private synchronized void initialize() {
86 m_simDevice = SimDevice.create("Ultrasonic", m_echoChannel.getChannel());
87 if (m_simDevice != null) {
88 m_simRangeValid = m_simDevice.createBoolean("Range Valid", Direction.kInput, true);
89 m_simRange = m_simDevice.createDouble("Range (in)", Direction.kInput, 0.0);
90 m_pingChannel.setSimDevice(m_simDevice);
91 m_echoChannel.setSimDevice(m_simDevice);
92 }
93 final boolean originalMode = m_automaticEnabled;
94 setAutomaticMode(false); // kill task when adding a new sensor
95 m_sensors.add(this);
96
97 m_counter = new Counter(m_echoChannel); // set up counter for this
98 SendableRegistry.addChild(this, m_counter);
99 // sensor
100 m_counter.setMaxPeriod(1.0);
101 m_counter.setSemiPeriodMode(true);
102 m_counter.reset();
103 m_enabled = true; // make it available for round-robin scheduling
104 setAutomaticMode(originalMode);
105
106 m_instances++;
107 HAL.report(tResourceType.kResourceType_Ultrasonic, m_instances);
108 SendableRegistry.addLW(this, "Ultrasonic", m_echoChannel.getChannel());
109 }

164 @Override
165 public synchronized void close() {
166 SendableRegistry.remove(this);
167 final boolean wasAutomaticMode = m_automaticEnabled;
168 setAutomaticMode(false);
169 if (m_allocatedChannels) {
170 if (m_pingChannel != null) {
171 m_pingChannel.close();
172 }
173 if (m_echoChannel != null) {
174 m_echoChannel.close();
175 }
176 }
177
178 if (m_counter != null) {
179 m_counter.close();
180 m_counter = null;
181 }
182
183 m_pingChannel = null;
184 m_echoChannel = null;
185 synchronized (m_sensors) {
186 m_sensors.remove(this);
187 }
188 if (!m_sensors.isEmpty() && wasAutomaticMode) {
189 setAutomaticMode(true);
190 }
191
192 if (m_simDevice != null) {
193 m_simDevice.close();
194 m_simDevice = null;
195 }
196 }

209 public static synchronized void setAutomaticMode(boolean enabling) {
210 if (enabling == m_automaticEnabled) {
211 return; // ignore the case of no change
212 }
213 m_automaticEnabled = enabling;
214
215 if (enabling) {
216 /* Clear all the counters so no data is valid. No synchronization is
217 * needed because the background task is stopped.
218 */
219 for (Ultrasonic u : m_sensors) {
220 u.m_counter.reset();
221 }
222
223 // Start round robin task
224 m_task = new UltrasonicChecker();
225 m_task.start();
226 } else {
227 if (m_task != null) {
228 // Wait for background task to stop running
229 try {
230 m_task.join();
231 m_task = null;
232 } catch (InterruptedException ex) {
233 Thread.currentThread().interrupt();
234 ex.printStackTrace();
235 }
236 }
237
238 /* Clear all the counters (data now invalid) since automatic mode is
239 * disabled. No synchronization is needed because the background task is
240 * stopped.
241 */
242 for (Ultrasonic u : m_sensors) {
243 u.m_counter.reset();
244 }
245 }
246 }

254 public void ping() {
255 setAutomaticMode(false); // turn off automatic round-robin if pinging
256 // single sensor
257 m_counter.reset(); // reset the counter to zero (invalid data now)
258 // do the ping to start getting a single range
259 m_pingChannel.pulse(kPingTime);

282 public double getRangeInches() {
283 if (isRangeValid()) {
284 if (m_simRange != null) {
285 return m_simRange.get();
286 }
287 return m_counter.getPeriod() * kSpeedOfSoundInchesPerSec / 2.0;
288 } else {
289 return 0;
290 }
291 }
