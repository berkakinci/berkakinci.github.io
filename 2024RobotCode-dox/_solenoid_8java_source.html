42 @SuppressWarnings("this-escape")
43 public Solenoid(final int module, final PneumaticsModuleType moduleType, final int channel) {
44 m_module = PneumaticsBase.getForType(module, moduleType);
45
46 m_mask = 1 << channel;
47 m_channel = channel;
48
49 if (!m_module.checkSolenoidChannel(channel)) {
50 m_module.close();
51 throw new IllegalArgumentException("Channel " + channel + " out of range");
52 }
53
54 if (m_module.checkAndReserveSolenoids(m_mask) != 0) {
55 m_module.close();
56 throw new AllocationException("Solenoid already allocated");
57 }
58
59 HAL.report(tResourceType.kResourceType_Solenoid, channel + 1, m_module.getModuleNumber() + 1);
60 SendableRegistry.addLW(this, "Solenoid", m_module.getModuleNumber(), channel);
61 }
