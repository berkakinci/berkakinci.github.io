74 headingHorizontal.getAsDouble(),
75 headingVertical.getAsDouble());

77 // Prevent Movement After Auto

87 }
88 //Dont Init Rotation Again
89 initRotation = false;
90 }
91
92 // Limit velocity to prevent tippy
93 Translation2d translation = SwerveController.getTranslation2d(desiredSpeeds);
94 translation = SwerveMath.limitVelocity(translation, swerve.getFieldVelocity(), swerve.getPose(),
95 Constants.LOOP_TIME, Constants.ROBOT_MASS, List.of(Constants.CHASSIS),
96 swerve.getSwerveDriveConfiguration());
97 SmartDashboard.putNumber("LimitedTranslation", translation.getX());
98 SmartDashboard.putString("Translation", translation.toString());
99
100 // Make the robot move
101 swerve.drive(translation, desiredSpeeds.omegaRadiansPerSecond, true);

103 }
