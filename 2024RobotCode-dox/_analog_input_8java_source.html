214 // Wait until the next sample, so the next call to getAccumulator*()
215 // won't have old values.
216 final double sampleTime = 1.0 / getGlobalSampleRate();
217 final double overSamples = 1 << getOversampleBits();
218 final double averageSamples = 1 << getAverageBits();
