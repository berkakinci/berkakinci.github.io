39 @Test
40 void solenoidOutputTest() {
41 HAL.initialize(500, 0);
42
43 try (PneumaticsControlModule pcm = new PneumaticsControlModule(0);
44 DoubleSolenoid doubleSolenoid = new DoubleSolenoid(0, PneumaticsModuleType.CTREPCM, 3, 4)) {
45 CTREPCMSim sim = new CTREPCMSim(0);
46 sim.resetData();
47
48 BooleanCallback callback3 = new BooleanCallback();
49 BooleanCallback callback4 = new BooleanCallback();
50
51 try (CallbackStore cb3 = sim.registerSolenoidOutputCallback(3, callback3, false);
52 CallbackStore cb4 = sim.registerSolenoidOutputCallback(4, callback4, false)) {
53 // Reverse
54 callback3.reset();
55 callback4.reset();
56 doubleSolenoid.set(DoubleSolenoid.Value.kReverse);
57 assertFalse(callback3.wasTriggered());
58 assertNull(callback3.getSetValue());
59 assertTrue(callback4.wasTriggered());
60 assertTrue(callback4.getSetValue());
61 assertFalse(sim.getSolenoidOutput(3));
62 assertTrue(sim.getSolenoidOutput(4));
63 assertEquals(0x10, pcm.getSolenoids());
64
65 // Forward
66 callback3.reset();
67 callback4.reset();
68 doubleSolenoid.set(DoubleSolenoid.Value.kForward);
69 assertTrue(callback3.wasTriggered());
70 assertTrue(callback3.getSetValue());
71 assertTrue(callback4.wasTriggered());
72 assertFalse(callback4.getSetValue());
73 assertTrue(sim.getSolenoidOutput(3));
74 assertFalse(sim.getSolenoidOutput(4));
75 assertEquals(0x8, pcm.getSolenoids());
76
77 // Off
78 callback3.reset();
79 callback4.reset();
80 doubleSolenoid.set(DoubleSolenoid.Value.kOff);
81 assertTrue(callback3.wasTriggered());
82 assertFalse(callback3.getSetValue());
83 assertFalse(callback4.wasTriggered());
84 assertNull(callback4.getSetValue());
85 assertFalse(sim.getSolenoidOutput(3));
86 assertFalse(sim.getSolenoidOutput(4));
87 assertEquals(0x0, pcm.getSolenoids());
88 }
89 }
