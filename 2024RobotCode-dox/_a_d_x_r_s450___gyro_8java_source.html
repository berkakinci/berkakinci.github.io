62 @SuppressWarnings("this-escape")
63 public ADXRS450_Gyro(SPI.Port port) {
64 m_spi = new SPI(port);
65
66 // simulation
67 m_simDevice = SimDevice.create("Gyro:ADXRS450", port.value);
68 if (m_simDevice != null) {
69 m_simConnected = m_simDevice.createBoolean("connected", SimDevice.Direction.kInput, true);
70 m_simAngle = m_simDevice.createDouble("angle_x", SimDevice.Direction.kInput, 0.0);
71 m_simRate = m_simDevice.createDouble("rate_x", SimDevice.Direction.kInput, 0.0);
72 }
73
74 m_spi.setClockRate(3000000);
75 m_spi.setMode(SPI.Mode.kMode0);
76 m_spi.setChipSelectActiveLow();
77
78 if (m_simDevice == null) {
79 // Validate the part ID
80 if ((readRegister(kPIDRegister) & 0xff00) != 0x5200) {
81 m_spi.close();
82 m_spi = null;
83 DriverStation.reportError("could not find ADXRS450 gyro on SPI port " + port.value, false);
84 return;
85 }
86
87 m_spi.initAccumulator(
88 kSamplePeriod, 0x20000000, 4, 0x0c00000e, 0x04000000, 10, 16, true, true);
89
90 calibrate();
91 }
92
93 HAL.report(tResourceType.kResourceType_ADXRS450, port.value + 1);
94 SendableRegistry.addLW(this, "ADXRS450_Gyro", port.value);

142 private boolean calcParity(int value) {
143 boolean parity = false;
144 while (value != 0) {
145 parity = !parity;
146 value = value & (value - 1);
147 }
148 return parity;
149 }

151 private int readRegister(int reg) {
152 int cmdhi = 0x8000 | (reg << 1);
153 boolean parity = calcParity(cmdhi);
154
155 ByteBuffer buf = ByteBuffer.allocate(4);
156 buf.order(ByteOrder.BIG_ENDIAN);
157 buf.put(0, (byte) (cmdhi >> 8));
158 buf.put(1, (byte) (cmdhi & 0xff));
159 buf.put(2, (byte) 0);
160 buf.put(3, (byte) (parity ? 0 : 1));
161
162 m_spi.write(buf, 4);
163 m_spi.read(false, buf, 4);
164
165 if ((buf.get(0) & 0xe0) == 0) {
166 return 0; // error, return 0
167 }
168 return (buf.getInt(0) >> 5) & 0xffff;
169 }
