324 public enum RadioLEDState {
326 kOff(LEDJNI.RADIO_LED_STATE_OFF),
328 kGreen(LEDJNI.RADIO_LED_STATE_GREEN),
330 kRed(LEDJNI.RADIO_LED_STATE_RED),
332 kOrange(LEDJNI.RADIO_LED_STATE_ORANGE);
333
335 public final int value;
336
337 RadioLEDState(int value) {
338 this.value = value;
339 }
340
347 public static RadioLEDState fromValue(int value) {
348 switch (value) {
349 case LEDJNI.RADIO_LED_STATE_OFF:
350 return RadioLEDState.kOff;
351 case LEDJNI.RADIO_LED_STATE_GREEN:
352 return RadioLEDState.kGreen;
353 case LEDJNI.RADIO_LED_STATE_RED:
354 return RadioLEDState.kRed;
355 case LEDJNI.RADIO_LED_STATE_ORANGE:
356 return RadioLEDState.kOrange;
357 default:
358 return RadioLEDState.kOff;
359 }
360 }
361 }
