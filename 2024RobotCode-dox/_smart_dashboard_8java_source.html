67 @SuppressWarnings("PMD.CompareObjectsWithEquals")
68 public static synchronized void putData(String key, Sendable data) {
69 if (!m_reported) {
70 HAL.report(tResourceType.kResourceType_SmartDashboard, tInstances.kSmartDashboard_Instance);
71 m_reported = true;
72 }
73 Sendable sddata = tablesToData.get(key);
74 if (sddata == null || sddata != data) {
75 tablesToData.put(key, data);
76 NetworkTable dataTable = table.getSubTable(key);
77 SendableBuilderImpl builder = new SendableBuilderImpl();
78 builder.setTable(dataTable);
79 SendableRegistry.publish(data, builder);
80 builder.startListeners();
81 dataTable.getEntry(".name").setString(key);
82 }
83 }
