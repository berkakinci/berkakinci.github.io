55 public SwerveSubsystem(File directory) {
56 // Angle conversion factor is 360 / (GEAR RATIO * ENCODER RESOLUTION)
57 // In this case the gear ratio is 12.8 motor revolutions per wheel rotation.
58 // The encoder resolution per motor revolution is 1 per motor revolution.
59 double angleConversionFactor = SwerveMath.calculateDegreesPerSteeringRotation(12.8, 1);
60 // Motor conversion factor is (PI * WHEEL DIAMETER IN METERS) / (GEAR RATIO * ENCODER RESOLUTION).
61 // In this case the wheel diameter is 4 inches, which must be converted to meters to get meters/second.
62 // The gear ratio is 6.75 motor revolutions per wheel rotation.
63 // The encoder resolution per motor revolution is 1 per motor revolution.
64 double driveConversionFactor = SwerveMath.calculateMetersPerRotation(Units.inchesToMeters(4), 6.75, 1);
65 System.out.println("\"conversionFactor\": {");
66 System.out.println("\t\"angle\": " + angleConversionFactor + ",");
67 System.out.println("\t\"drive\": " + driveConversionFactor);
68 System.out.println("}");
69
70 // Configure the Telemetry before creating the SwerveDrive to avoid unnecessary objects being created.
71 SwerveDriveTelemetry.verbosity = TelemetryVerbosity.HIGH;
72 try {
73 swerveDrive = new SwerveParser(directory).createSwerveDrive(maximumSpeed);
74 // Alternative method if you don't want to supply the conversion factor via JSON files.
75 // swerveDrive = new SwerveParser(directory).createSwerveDrive(maximumSpeed, angleConversionFactor, driveConversionFactor);
76 } catch (Exception e) {
77 throw new RuntimeException(e);
78 }
79 swerveDrive.setHeadingCorrection(false); // Heading correction should only be used while controlling the robot via angle.
80
81 AutoBuilder.configureHolonomic(
82 this::getPose,
83 this::resetOdometry,
84 this::getRobotVelocity,
85 this::drive,
86 Constants.pathFollowerConfig,
87 () -> {
88 // Boolean supplier that controls when the path will be mirrored for the red alliance
89 // This will flip the path being followed to the red side of the field.
90 // THE ORIGIN WILL REMAIN ON THE BLUE SIDE
91
92 var alliance = DriverStation.getAlliance();
93 if (alliance.isPresent()) {
94 return alliance.get() == DriverStation.Alliance.Red;
95 }
96 return false;
97 },
98 this
99 );
100
101 //logging
102 PathPlannerLogging.setLogActivePathCallback((poses) -> swerveDrive.field.getObject("path").setPoses(poses));
103
104 SmartDashboard.putData("Field", swerveDrive.field);
105
106 }
