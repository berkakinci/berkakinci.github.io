59 public static synchronized void setNetworkTableInstance(NetworkTableInstance inst) {
60 m_table = inst.getTable(TABLE_NAME);
61 if (m_typePublisher != null) {
62 m_typePublisher.close();
63 }
64 m_typePublisher = m_table.getStringTopic(".type").publish();
65 m_typePublisher.set("RobotPreferences");
66
67 // Subscribe to all Preferences; this ensures we get the latest values
68 // ahead of a getter call.
69 if (m_tableSubscriber != null) {
70 m_tableSubscriber.close();
71 }
72 m_tableSubscriber = new MultiSubscriber(inst, new String[] {m_table.getPath() + "/"});
73
74 // Listener to set all Preferences values to persistent
75 // (for backwards compatibility with old dashboards).
76 if (m_listener != null) {
77 m_listener.close();
78 }
79 m_listener =
80 NetworkTableListener.createListener(
81 m_tableSubscriber,
82 EnumSet.of(NetworkTableEvent.Kind.kImmediate, NetworkTableEvent.Kind.kPublish),
83 event -> {
84 if (event.topicInfo != null) {
85 Topic topic = event.topicInfo.getTopic();
86 if (!topic.equals(m_typePublisher.getTopic())) {
87 event.topicInfo.getTopic().setPersistent(true);
88 }
89 }
90 });
91 }
