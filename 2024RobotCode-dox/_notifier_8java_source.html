19public class Notifier implements AutoCloseable {

23 // The lock held while updating process information.

26 // HAL handle passed to the JNI bindings (atomic for proper destruction).

29 // The user-provided callback.

33 // zero as RobotController.getFPGATime().

43 @Override
44 public void close() {
45 int handle = m_notifier.getAndSet(0);
46 if (handle == 0) {
47 return;
48 }
49 NotifierJNI.stopNotifier(handle);
50 // Join the thread to ensure the callback has exited.
51 if (m_thread.isAlive()) {
52 try {
53 m_thread.interrupt();
54 m_thread.join();
55 } catch (InterruptedException ex) {
56 Thread.currentThread().interrupt();
57 }
58 }
59 NotifierJNI.cleanNotifier(handle);
60 m_thread = null;
61 }
62
68 private void updateAlarm(long triggerTimeMicroS) {
69 int notifier = m_notifier.get();
70 if (notifier == 0) {
71 return;
72 }
73 NotifierJNI.updateNotifierAlarm(notifier, triggerTimeMicroS);
74 }
75
77 private void updateAlarm() {
78 updateAlarm((long) (m_expirationTimeSeconds * 1e6));
79 }

88 public Notifier(Runnable callback) {
89 requireNonNullParam(callback, "callback", "Notifier");
90
91 m_callback = callback;
92 m_notifier.set(NotifierJNI.initializeNotifier());
93
94 m_thread =
95 new Thread(
96 () -> {
97 while (!Thread.interrupted()) {
98 int notifier = m_notifier.get();
99 if (notifier == 0) {
100 break;
101 }
102 long curTime = NotifierJNI.waitForNotifierAlarm(notifier);
103 if (curTime == 0) {
104 break;
105 }
106
107 Runnable threadHandler;
108 m_processLock.lock();
109 try {
110 threadHandler = m_callback;
111 if (m_periodic) {
112 m_expirationTimeSeconds += m_periodSeconds;
113 updateAlarm();
114 } else {
115 // Need to update the alarm to cause it to wait again
116 updateAlarm(-1);
117 }
118 } finally {
119 m_processLock.unlock();
120 }
121
122 // Call callback
123 if (threadHandler != null) {
124 threadHandler.run();
125 }
126 }
127 });
128 m_thread.setName("Notifier");
129 m_thread.setDaemon(true);
130 m_thread.setUncaughtExceptionHandler(
131 (thread, error) -> {
132 Throwable cause = error.getCause();
133 if (cause != null) {
134 error = cause;
135 }
136 DriverStation.reportError(
137 "Unhandled exception in Notifier thread: " + error, error.getStackTrace());
138 DriverStation.reportError(
139 "The Runnable for this Notifier (or methods called by it) should have handled "
140 + "the exception above.\n"
141 + " The above stacktrace can help determine where the error occurred.\n"
142 + " See https://wpilib.org/stacktrace for more information.",
143 false);
144 });
145 m_thread.start();
146 }
