32 ShuffleboardInstance(NetworkTableInstance ntInstance) {
33 requireNonNullParam(ntInstance, "ntInstance", "ShuffleboardInstance");

40 @Override
41 public ShuffleboardTab getTab(String title) {
42 requireNonNullParam(title, "title", "getTab");
43 if (!m_reported) {
44 HAL.report(tResourceType.kResourceType_Shuffleboard, 0);
45 m_reported = true;
46 }
47 if (!m_tabs.containsKey(title)) {
48 m_tabs.put(title, new ShuffleboardTab(this, title));
49 m_tabsChanged = true;
50 }
51 return m_tabs.get(title);
52 }
53
54 @Override
55 public void update() {
56 if (m_tabsChanged) {
57 String[] tabTitles =
58 m_tabs.values().stream().map(ShuffleboardTab::getTitle).toArray(String[]::new);
59 m_rootMetaTable.getEntry("Tabs").setStringArray(tabTitles);
60 m_tabsChanged = false;
61 }
62 for (ShuffleboardTab tab : m_tabs.values()) {
63 String title = tab.getTitle();
64 tab.buildInto(m_rootTable, m_rootMetaTable.getSubTable(title));
65 }
66 }

104 private void apply(ShuffleboardContainer container, Consumer<ComplexWidget> func) {
105 for (ShuffleboardComponent<?> component : container.getComponents()) {
106 if (component instanceof ComplexWidget) {
107 func.accept((ComplexWidget) component);
108 }
109 if (component instanceof ShuffleboardContainer) {
110 apply((ShuffleboardContainer) component, func);
