37 @SuppressWarnings("this-escape")
38 public Tachometer(DigitalSource source) {
39 m_source = requireNonNullParam(source, "source", "Tachometer");
40
41 ByteBuffer index = ByteBuffer.allocateDirect(4);
42 // set the byte order
43 index.order(ByteOrder.LITTLE_ENDIAN);
44 m_handle = CounterJNI.initializeCounter(CounterJNI.TWO_PULSE, index.asIntBuffer());
45
46 CounterJNI.setCounterUpSource(
47 m_handle, source.getPortHandleForRouting(), source.getAnalogTriggerTypeForRouting());
48 CounterJNI.setCounterUpSourceEdge(m_handle, true, false);
49
50 int intIndex = index.getInt();
51 HAL.report(tResourceType.kResourceType_Counter, intIndex + 1);
52 SendableRegistry.addLW(this, "Tachometer", intIndex);
53 }
