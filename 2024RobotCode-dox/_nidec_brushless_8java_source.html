34 @SuppressWarnings("this-escape")
35 public NidecBrushless(final int pwmChannel, final int dioChannel) {
36 setSafetyEnabled(false);
37
38 // the dio controls the output (in PWM mode)
39 m_dio = new DigitalOutput(dioChannel);
40 SendableRegistry.addChild(this, m_dio);
41 m_dio.setPWMRate(15625);
42 m_dio.enablePWM(0.5);
43
44 // the pwm enables the controller
45 m_pwm = new PWM(pwmChannel);
46 SendableRegistry.addChild(this, m_pwm);
47
48 HAL.report(tResourceType.kResourceType_NidecBrushless, pwmChannel + 1);
49 SendableRegistry.addLW(this, "Nidec Brushless", pwmChannel);
50 }
51
52 @Override
53 public void close() {
54 SendableRegistry.remove(this);
55 m_dio.close();
56 m_pwm.close();
57 }

67 @Override

69 if (!m_disabled) {
