46 public void setHSV(final int index, final int h, final int s, final int v) {
47 if (s == 0) {
48 setRGB(index, v, v, v);
49 return;
50 }
51
52 // The below algorithm is copied from Color.fromHSV and moved here for
53 // performance reasons.
54
55 // Loosely based on
56 // https://en.wikipedia.org/wiki/HSL_and_HSV#HSV_to_RGB
57 // The hue range is split into 60 degree regions where in each region there
58 // is one rgb component at a low value (m), one at a high value (v) and one
59 // that changes (X) from low to high (X+m) or high to low (v-X)
60
61 // Difference between highest and lowest value of any rgb component
62 final int chroma = (s * v) / 255;
63
64 // Because hue is 0-180 rather than 0-360 use 30 not 60
65 final int region = (h / 30) % 6;
66
67 // Remainder converted from 0-30 to 0-255
68 final int remainder = (int) Math.round((h % 30) * (255 / 30.0));
69
70 // Value of the lowest rgb component
71 final int m = v - chroma;
72
73 // Goes from 0 to chroma as hue increases
74 final int X = (chroma * remainder) >> 8;
75
76 switch (region) {
77 case 0:
78 setRGB(index, v, X + m, m);
79 break;
80 case 1:
81 setRGB(index, v - X, v, m);
82 break;
83 case 2:
84 setRGB(index, m, v, X + m);
85 break;
86 case 3:
87 setRGB(index, m, v - X, v);
88 break;
89 case 4:
90 setRGB(index, X + m, m, v);
91 break;
92 default:
93 setRGB(index, v, m, v - X);
94 break;
95 }
96 }
